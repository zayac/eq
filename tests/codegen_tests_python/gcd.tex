\begin{eqcode}{gcd}{m, n}{\type{Z}, \type{Z}}{\type{Z}}
  \qif {m = 0}
    \return {n} \lend
  \qelse
    \return {\call{gcd}{n \mod m, m}} \lend
  \qendif
\end{eqcode}
